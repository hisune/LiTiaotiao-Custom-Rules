```
{"popup_rules":
    [
        {"id":"tv_upgrade_title","action":"iv_upgrade_btn_ignore"},
        {"id":"广告弹窗","action":"关闭弹窗"},
        {"id":"共享&广告标识","action":"不允许"},
        {"id":"开启通知提醒","action":"im_feed_tip_close"}

```
